%% \head
% A typical ``head of slide'' macro, but anything can be used.
% This one makes a table of contents entry
\def\head#1{
\clearpage
 \begin{center}~\\
 \Large\sf\black{#1}\\
 \color[rgb]{0.4,0.9,0.3} \rule{\textwidth}{1mm}
 \end{center}
 \usecounter{section}
 \addcontentsline{toc}{section}{\thepart\hspace{1em}#1}
}

%% User-level embedding macros.  
% Basically one provides a TeX ``box'' (typically using \epsfbox)
% and a movie object, which is set to the same size in the final
% output as the TeX box.

% \embed{640\px}{480\px}{movie.mpg}{movie.eps}
\newcommand{\embed}[4]{\awfembedmovie{\epsfxsize=#1\epsfysize=#2\epsfbox{#4}}{#3}}

% \embedwide{0.7\textwidth}{movie.mpg}{movie.eps}
\newcommand{\embedwide}[3]{\awfembedmovie{\epsfxsize=#1\epsfbox{#2}}{#3}}

% Embed h movie eps
\newcommand{\embedhigh}[3]{\awfembedmovie{\epsfxsize=#1\epsfbox{#2}}{#3}}

% awfembed{printed-object}{html-code}
\newcommand{\awfembed}[2]{{%
\setbox1=\hbox{#1}% Store #1 in a box
\newdimen\w\w=\wd1% Compute its width/height, store in \w, \h
\newdimen\h\h=\ht1
\mbox{\awfincludeps{
(#2) % html code to be put in <DIV>, of size: 
(\the\w) awfstrtonum   % Width in format ``1234.5pt'', convert to PostScript units
(\the\h) awfstrtonum  % Height, similarly
curposdiv }
\box1}% actualy send printed-object to TeX
}}
 
% awfembedmovie{textobject}{movie.mpg}
\newcommand{\awfembedmovie}[2]{
\awfembed{\mbox{#1}}{<embed
controller="false"
playcount=3   for ie
loop=3 for netscape
% framerate=10 
autorewind="false"
TransparentAtStart=true
autostart="true"
appearance="1"
borderstyle="0"
src=#2 
WIDTH=$WIDTH$ HEIGHT=$HEIGHT$>}}


%% \textalkdata{title}{author}{date}
% Set these variables in the postscript file, so textalk-header
% can use them for filenames, page titles etc.
\def\textalktitle#1#2#3{%
\awfincludepsheader{/title (#1) def}
\awfincludepsheader{/author (#2) def}
\awfincludepsheader{/date (#3) def}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Internals: Code which speaks to textalk-header.ps

% Pass text directly to dvips, and hence to PostScript.
\def\textalkspecial#1{%
  \begingroup
    \edef~{\string~}%
    %... other redefinitions if needed.
    \special{#1}%
  \endgroup
}

% Evaluate postscript in the file header, within textalkdict if
% known, otherwise blank.
\def\awfincludepsheader#1{\textalkspecial{!/textalkdict where { pop
textalkdict begin
#1
end
} {} ifelse }}


\def\awfincludeps#1{\textalkspecial{"/textalkdict where { pop
textalkdict begin
#1
end
} {} ifelse }}

\def\awfincludehtml#1{\awfincludeps{
(#1) dup hprintnl
(awfincludehtml:[) print print (]) print
}}
